  <script src="https://cdn.jsdelivr.net/npm/glightbox/dist/js/glightbox.min.js"></script>
  <script>
    GLightbox({ selector: '.glightbox', touchNavigation: true, loop: true, descPosition: 'none' });
    const hero    = document.getElementById('projectHero');
    const heroImg = hero.querySelector('.project-hero-img');
    const overlay = document.getElementById('heroOverlay');
    const bar     = document.getElementById('heroBar');
    const textCol = document.querySelector('.project-text-col');
    const slides  = document.querySelectorAll('.project-slide');
    const dots    = document.getElementById('slideDots');
    const HERO_H0  = window.innerHeight * 0.6;
    const BAR_H    = 52;
    const COLLAPSE = HERO_H0 - BAR_H;
    textCol.style.paddingTop = HERO_H0 + 'px';
    let current = 0;
    slides.forEach((_, i) => {
      const d = document.createElement('button');
      d.className = 'slide-dot' + (i === 0 ? ' active' : '');
      d.addEventListener('click', () => goTo(i));
      dots.appendChild(d);
    });
    function goTo(n) {
      slides[current].classList.remove('active');
      dots.children[current].classList.remove('active');
      current = (n + slides.length) % slides.length;
      slides[current].classList.add('active');
      dots.children[current].classList.add('active');
    }
    function lerp(a, b, t) { return a + (b - a) * t; }
    function clamp01(x)    { return Math.min(1, Math.max(0, x)); }
    function onTextScroll() {
      const s = textCol.scrollTop;
      const p = clamp01(s / COLLAPSE);
      const h = lerp(HERO_H0, BAR_H, p);
      hero.style.height        = h + 'px';
      textCol.style.paddingTop = h + 'px';
      heroImg.style.opacity    = lerp(0.45, 0, clamp01(p / 0.7));
      const op = clamp01(p / 0.5);
      overlay.style.opacity   = 1 - op;
      overlay.style.transform = `translateY(${-op * 20}px)`;
      const bp = clamp01((p - 0.6) / 0.4);
      bar.style.opacity = bp;
      bar.classList.toggle('active', bp > 0);
      if (slides.length > 1) {
        const maxScroll = textCol.scrollHeight - textCol.clientHeight;
        const sp = maxScroll > COLLAPSE ? clamp01((s - COLLAPSE) / (maxScroll - COLLAPSE)) : 0;
        const idx = Math.min(Math.floor(sp * slides.length), slides.length - 1);
        if (idx !== current) goTo(idx);
      }
    }
    textCol.addEventListener('scroll', onTextScroll, { passive: true });
    onTextScroll();
  </script>
</body>
</html>
